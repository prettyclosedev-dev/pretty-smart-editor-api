          const response = await fetch(apiUrl, {

            },

              variables,
            }),
          });

          const data = await response.json();
          if (data && data.data && data.data.unsplash && !data.data.unsplash.errors && data.data.unsplash.results && data.data.unsplash.results.length && data.data.unsplash.results[0].urls && data.data.unsplash.results[0].urls.regular) {

function fallBackImg(elem) {
  elem.error = null;
  let index = elem.dataset.fallIndex || 1;
  const maxFallbackImages = 2; // Update this to the maximum number of fallback images you have

  if (index > maxFallbackImages) {
    console.error("All fallback images failed to load.");
    return;
  }

  elem.src = elem.dataset[`src-${index}`];
  elem.dataset.fallIndex = ++index;
}
